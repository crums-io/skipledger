Source-pack Binary Format
=========================

Rows are modeled as a list of cells. Each cell incurs at least one byte of
overhead, which is needed to distinguish redacted cells from revealed ones.
This CELL_CODE is also used to encode the cell's data type. (Note the corner-case
type NULL is meant to represent SQL NULLs, but does not distinguish which type
was set to null. Its value is a zeroed 5-byte buffer that is never written to
the stream -- the type code alone marks it.)


// notation definition
array(T)        := SHORT T ^ SHORT    // 2-byte unsigned count, followed by count-many T's



SCHEMA_FLAG     := LONG               // 8-byte bit field
                                      //
                                      //  bit 0 (0x01)  -- iso-count: all rows have same cell count
                                      //
                                      //  bits 1-2 (0x06) encode the salt scheme:
                                      //    0x00  -- not salted  (no index array)
                                      //    0x02  -- salted cell indices present
                                      //    0x04  -- unsalted exception indices present (all others salted)
                                      //    0x06  -- all salted  (no index array)


// present iff SCHEMA_FLAG & 6 == 2
SALTED_CELLS    := array(SHORT)       // ascending cell indices that ARE salted

// present iff SCHEMA_FLAG & 6 == 4
UNSALTED_CELLS  := array(SHORT)       // ascending cell indices that are NOT salted


// present iff SCHEMA_FLAG & 1 == 1 (iso-count)
ISO_COUNT       := BYTE               // fixed cell-count for every row (1 byte; max 255)

// present iff SCHEMA_FLAG & 1 == 0
CELL_COUNT_SIZE := BYTE               // no. of bytes used to encode per-row cell count
                                      // valid values: 1, 2, 3


VAR_SIZE_LEN    := BYTE               // no. of bytes used to encode the size of var-size types
                                      // valid values: 2, 3, 4


HEADER  := SCHEMA_FLAG
           [SALTED_CELLS | UNSALTED_CELLS]
           <ISO_COUNT | CELL_COUNT_SIZE>
           VAR_SIZE_LEN


ROW_COUNT       := INT                // big-endian 4-byte value; max 0x7f_ff_ff_ff



// ── Per-row fields, in ascending order of row number ──────────────────────────

NO              := LONG               // 1-based row number

// present iff SCHEMA_FLAG & 1 == 0  (not iso-count)
CC              := BYTE ^ CELL_COUNT_SIZE
                                      // unsigned big-endian integer; 1, 2, or 3 bytes wide

ROW_STATUS      := BYTE               // bit field
                                      //  bit 0 (0x01) -- 1 = row salt present
                                      //                  0 = no row salt (inline cell salts used,
                                      //                      or no salt at all)
                                      //  bit 1 (0x02) -- reserved (whitespace-tokenized, TODO)

// present iff ROW_STATUS & 1 == 1  AND  SCHEMA_FLAG & 6 != 0
ROW_SALT        := BYTE ^ 32          // 32-byte root salt; all cell salts are derived from this



// ── Per-cell fields ────────────────────────────────────────────────────────────

CELL_CODE       := BYTE               // encodes cell type and whether the cell is redacted
                                      //  0  -- redacted  (only the cell hash is stored)
                                      //  1  -- NULL    (no data written; 5-byte zeroed value implied)
                                      //  2  -- STRING  (UTF-8, var size)
                                      //  3  -- LONG    (big-endian signed 8-byte integer)
                                      //  4  -- BIG_INT (two's-complement byte array, var size; >= 1 byte)
                                      //  5  -- BIG_DEC (1-byte scale + BIG_INT unscaled value, var size; >= 2 bytes)
                                      //  6  -- DATE    (UTC milliseconds, big-endian signed 8-byte integer)
                                      //  7  -- BOOL    (1 byte: 0=false, 1=true)
                                      //  8  -- BYTES   (untyped blob, var size)
                                      //  9  -- HASH    (32-byte SHA-256 digest)
                                      //
                                      // CELL_CODE == DataType.ordinal() + 1  for unredacted cells


// present iff CELL_CODE indicates a var-size type (STRING, BIG_INT, BIG_DEC, BYTES)
CELL_DATA_SIZE  := BYTE ^ VAR_SIZE_LEN
                                      // unsigned big-endian integer; 2, 3, or 4 bytes wide

// present iff ROW_STATUS & 1 == 0  AND  CELL_CODE != 0  AND  cell index is salted
// (when ROW_STATUS & 1 == 1, the cell salt is derived from ROW_SALT -- not stored inline)
CELL_SALT       := BYTE ^ 32

// present iff CELL_CODE != 0  (i.e. not redacted)  AND  CELL_CODE != 1  (i.e. not NULL)
CELL_DATA       := BYTE ^ CELL_DATA_SIZE


// present iff CELL_CODE == 0  (redacted)
CELL_HASH       := BYTE ^ 32


CELL    := CELL_CODE
           [CELL_DATA_SIZE]
           [CELL_SALT]
           <CELL_DATA | CELL_HASH>

ROW     := NO [CC] ROW_STATUS [ROW_SALT] CELL ^ <CC | ISO_COUNT>

SRC_PACK := HEADER ROW_COUNT ROW ^ ROW_COUNT
